269  template <typename PointerType>
270  static void HandleWeak(const LivenessBroker& info, const void* object) {
271  const PointerType* weak = static_cast<const PointerType*>(object);
272  // Sentinel values are preserved for weak pointers.
273  if (*weak == kSentinelPointer) return;
274  const auto* raw = weak->Get();
275  if (!info.IsHeapObjectAlive(raw)) {
276  weak->ClearFromGC();
277  }
278  }
