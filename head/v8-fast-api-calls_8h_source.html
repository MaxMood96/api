208  enum class ArgFlags : uint8_t {
209  kNone = 0,
210  kIsArrayBit = 1 << 0, // This argument is first in an array of values.
211  };

213  static CTypeInfo FromWrapperType(ArgFlags flags = ArgFlags::kNone) {
214  return CTypeInfo(static_cast<int>(flags) | kIsWrapperTypeBit);
215  }

217  static constexpr CTypeInfo FromCType(Type ctype,
218  ArgFlags flags = ArgFlags::kNone) {
219  // TODO(mslekova): Refactor the manual bit manipulations to use
220  // PointerWithPayload instead.
221  // ctype cannot be Type::kV8Value.
222  return CTypeInfo(
223  ((static_cast<uintptr_t>(ctype) << kTypeOffset) & kTypeMask) |
224  static_cast<int>(flags));
225  }
226 
227  const void* GetWrapperInfo() const;
228 
229  constexpr Type GetType() const {
230  if (payload_ & kIsWrapperTypeBit) {
231  return Type::kV8Value;
232  }
233  return static_cast<Type>((payload_ & kTypeMask) >> kTypeOffset);
234  }
235 
236  constexpr bool IsArray() const {
237  return payload_ & static_cast<int>(ArgFlags::kIsArrayBit);
238  }
239 
240  static const CTypeInfo& Invalid() {
241  static CTypeInfo invalid = CTypeInfo(0);
242  return invalid;
243  }
244 
245  private:
246  explicit constexpr CTypeInfo(uintptr_t payload) : payload_(payload) {}
247 
248  // That must be the last bit after ArgFlags.
249  static constexpr uintptr_t kIsWrapperTypeBit = 1 << 1;
250  static constexpr uintptr_t kWrapperTypeInfoMask = static_cast<uintptr_t>(~0)
251  << 2;
252 
253  static constexpr unsigned int kTypeOffset = kIsWrapperTypeBit;
254  static constexpr unsigned int kTypeSize = 8 - kTypeOffset;
255  static constexpr uintptr_t kTypeMask =
256  (~(static_cast<uintptr_t>(~0) << kTypeSize)) << kTypeOffset;
257 
258  const uintptr_t payload_;
259 };
